    FILE(GLOB DLLS

    INSTALL(FILES ${PYDFILES} DESTINATION calimero)
    INSTALL(FILES ${MANIFEST} DESTINATION bin)


    SET(CPACK_NSIS_EXTRA_INSTALL_COMMANDS "
        WriteRegStr HKCU \\\"Software\\\\IUT\\\\Calimero\\\" calimerohome \\\"$INSTDIR\\\\lib\\\\,$INSTDIR\\\\calimero\\\\,$INSTDIR\\\\calimero\\\\site-packages\\\\\\\"

        ")
